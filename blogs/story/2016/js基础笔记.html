求任意数值中的最大值

function maxArr(arr = []) {
  // (1)先准备一个max变量存放数组的第一个值
  let maxarr = arr[0];
  // (2)冒泡遍历比较
  for (let index = 1; index < arr.length; index++) {
    if (maxarr < arr[index]) {
      maxarr = arr[index];
    }
  }
  // (3)返回
  return maxarr;
}
let max = maxArr([10, 500, 12, 15, 1]);
document.write(max);

// tip  计算时间公式
// 小时 h=parseInt(总秒数/60/60%24)
// 分支 m=parseInt(总秒数/60%60)
// 秒   s=parseInt(总秒数%60)
// 1、用户输入
let num = +prompt("请输入秒数...");
// 2、封装函数
function times() {
  h = parseInt((num / 60 / 60) % 24);
  m = parseInt((num / 60) % 60);
  s = parseInt(num % 60);
  //在前面多加一个0
  // h = h < 10 ? '0' + h : h
  // m = m < 10 ? '0' + m : m
  // s = s < 10 ? '0' + s : s
  return `${h}小时${m}分${s}秒`;
}
// 3、显示结果
let end = times(num);
document.write(end);

# 逻辑中断

  • 短路：只存在于&&和||中，当满足一定条件会让右边的代码不执行
符号 短路条件
&& 左边为 false 就短路
|| 左边为 true 就短路
  • 原因：通过左边能得到整个式子的结果，因此没必要在判断右边
function fu(x, y) {
  x = x || 0;
  y = y || 0;
  return x + y;
}
console.log(fu(1, 1)); // 结果为 2
console.log(fu(1)); // 结果为 1

# 随机生成 N~M 之间的整数

//万能取整数
function getRandom(N, M) {
  return Math.floor(Math.random() * (M - N + 1)) + N;
}

注：这个公式可以直接使用，取任意两位数之间的整数。

# 事件冒泡

  1. 事件冒泡概念：

    当一个元素的事件被触发时，同样的事件将会在该元素的所有祖先元素中依次被触发。这一过程就称为事件冒泡。(从下往上执行)

    • 简单理解：当一个元素触发事件后，以此向上调用所有父级元素的==同名事件==。(事件冒泡默认存在)

  2. 如何阻止冒泡？

    //  语法
事件对象.stopPropagation();

  3. 如何解绑事件?

    //	语法
对象.removeEventListenter("click", 函数名);

    attention：匿名函数无法被解绑

    function time() {
      const date = +new Date();
      const last = +new Date("2023/4/15 17:49");
      const shijian = (last - date) / 1000;
      // console.log(typeof shijian);
      let h = parseInt((shijian / 60 / 60) % 24);
      if (h <= 0) {
        h = "0" + 0;
      } else if (0 < h < 10) {
        h = "0" + h;
      }
      let m = parseInt((shijian / 60) % 60);
      if (m <= 0) {
        m = "0" + 0;
      } else if (m < 10) {
        m = "0" + m;
      }
      let s = parseInt(shijian % 60);
      if (s <= 0) {
        s = "0" + 0;
      } else if (s < 10) {
        s = "0" + s;
      } else {
        s = s;
      }

      document.querySelector(".hour").innerHTML = h;
      document.querySelector(".mit").innerHTML = m;
      document.querySelector(".miao").innerHTML = s;
    }

    setInterval(function() {
      time();
    }, 1000);
  </script>

# 字符类

预定类 说明
\d 匹配 0-9 之间的任一数字，相当[0-9]
\D 匹配所有 0-9 以外的字符，相当[^0-9]
\w 匹配任意的字母，数字和下划线，相当于[a-zA-Z0-9-_]
\W 除了所有字母、数字和下划线以外的字符，相当于[^a-za-z0-9-_]
\s 匹配空格(包括换行符、制表符、空格符等)，相当于[\t\r\n\v\f]
\S 匹配非空格符，相当于[^\t\r\n\v\f]
日期格式：^\d{4}-\d{1,2}-\d{1,2}
